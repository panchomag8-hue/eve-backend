async function sendMessage() { const input = document.getElementById("input"); const message = input.value; if (!message) return; const res = await fetch("/api/eve", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ message }) }); const data = await res.json(); addMessage("You", message); addMessage("Eve", data.reply); input.value = ""; }
